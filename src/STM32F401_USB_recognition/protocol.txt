パケットの形式 < 0xAA 0x55   0x00     0x00          ...          0xWW 0xXX 0xYY 0xZZ>
                { 接頭辞 } {タイプ} { 長さ } { データ(自由形式) } {checksum(CRC32)}

// タイプ(256種類まで追加可能)
#define RET      0x00  // 応答タイプ
#define CMD      0x01  // 命令タイプ


// 命令一覧
#define RECOGNIZE 0x00
#define GET       0x01

// 引数一覧
#define USB_TYPE 0x00
#define IMG      0x01

<Buffer 0xAA 0x55 0x01 0x02 0x00 0x00 0x52 0xAB 0x21 0xD4> // RECOGNIZE USB_TYPE
<Buffer 0xAA 0x55 0x01 0x02 0x01 0x01 0x39 0x77 0xF9 0x02> // GET IMG
